SELECT b.id as bid, b.name, b.edition, b.publisher, c.name as course, b.price, b.detailInformation,GROUP_CONCAT(ba.author) as authors from Book as b JOIN BookAuthor as ba JOIN CourseBook as cb JOIN Course as c on b.id=ba.bookid and b.id=cb.bookid and c.bsid=cb.courseid WHERE b.id in (%s,%s,%s,%s,%s) GROUP BY bid,c.name;
